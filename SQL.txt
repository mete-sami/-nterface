--29) 1997 yilindaki aylik satiþlari sirala. Ocak - 500 gibi toplamda 12 satir olacak

--30) En deðerli MÜÞTERÝMÝN adi ve soyadi (orders,orderdetails,customers)

select c.ContactName , sum(od.UnitPrice * od.Quantity) Tutar  from Customers c 
inner join Orders o on o.CustomerID = c.CustomerID 
inner join [Order Details] od on od.OrderID = o.OrderID 
group by c.ContactName order by Tutar desc
